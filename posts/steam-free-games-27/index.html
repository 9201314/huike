{

        jQuery( '.account_table a' ).each( function( i, el )
        {
                var match = el.href.match( /javascript:RemoveFreeLicense\( ([0-9]+), '/ );
                
                if( match !== null )

                        ownedPackages[ +match[ 1 ] ] = true;

}());
